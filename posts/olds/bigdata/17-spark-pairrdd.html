def compute_contribs(urls, rank):
    """ ç»™urlsè®¡ç®—
    Args:
        urls: ç›®æ ‡urlç›¸é‚»çš„urlsé›†åˆ
        rank: ç›®æ ‡urlçš„å½“å‰rank

    """ æŠŠä¸€è¡Œurlåˆ‡åˆ†å¼€æ¥
    Args:

    """ è®¡ç®—å„ä¸ªpageçš„æ’å

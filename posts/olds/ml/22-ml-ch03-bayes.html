def train_nb0(train_mat, class_list):
    ''' æœ´ç´ è´å¶æ–¯è®­ç»ƒç®—æ³•ï¼ŒäºŒåˆ†ç±»é—®é¢˜
    Args:

            c1_word_count += sum(train_mat[i])
